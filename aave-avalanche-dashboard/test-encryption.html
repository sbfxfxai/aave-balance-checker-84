// Test encryption/decryption for non-custodial wallet flow async function testEncryption() { const privateKey = "0x1234567890abcdef1234567890abcdef1234567890abcdef1234567890abcdef"; const email = "test@example.com"; const paymentId = "payment-123456789"; // Test encryption (client-side) const encoder = new TextEncoder(); const keyMaterial = await window.crypto.subtle.importKey( 'raw', encoder.encode(email + paymentId), { name: 'PBKDF2' }, false, ['deriveBits', 'deriveKey'] ); const key = await window.crypto.subtle.deriveKey( { name: 'PBKDF2', salt: encoder.encode('tiltvault-salt'), iterations: 100000, hash: 'SHA-256', }, keyMaterial, { name: 'AES-GCM', length: 256 }, true, ['encrypt', 'decrypt'] ); const iv = window.crypto.getRandomValues(new Uint8Array(12)); const encrypted = await window.crypto.subtle.encrypt( { name: 'AES-GCM', iv }, key, encoder.encode(privateKey) ); const combined = new Uint8Array(iv.length + new Uint8Array(encrypted).length); combined.set(iv); combined.set(new Uint8Array(encrypted), iv.length); const encryptedBase64 = btoa(String.fromCharCode(...combined)); console.log('Encrypted:', encryptedBase64); // Test decryption (would happen client-side when user recovers wallet) const encryptedData = Uint8Array.from(atob(encryptedBase64), c => c.charCodeAt(0)); const decryptedIv = encryptedData.slice(0, 12); const encryptedDataOnly = encryptedData.slice(12); const decrypted = await window.crypto.subtle.decrypt( { name: 'AES-GCM', iv: decryptedIv }, key, encryptedDataOnly ); const decryptedPrivateKey = new TextDecoder().decode(decrypted); console.log('Decrypted:', decryptedPrivateKey); console.log('Match:', decryptedPrivateKey === privateKey); } // Run test testEncryption().catch(console.error);
